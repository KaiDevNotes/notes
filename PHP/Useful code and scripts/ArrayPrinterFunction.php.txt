<?php

    // Функция для рекурсивного обхода массива и 
    // вывода его на экран в удобочетаемом виде 

    function arrayPrinter($a){
        $r = "<ul>";        
        foreach ($a as $k => $v){
            if (is_array($v)){
                $r .= "<li><strong>[".$k."] ::</strong>".arrayPrinter($v)."</li>";
            }
            else {
                $r .= "<li><strong>[".$k."] ---></strong> ".$v."</li>";
            }
        } 
        $r .= "</ul>";
        return $r;
    }
    
    
    //------------------------------------------
    // Пример использования:
    
    $array = array(
        "a" => "1",
        "b" => array(
            "b11" => "2",
            "b12" => array(
                "b121" => "3",
                "b122" => "3",
                "b123" => "3",
                "b124" => array(
                    "b1241" => "4",
                    "b1242" => "4",
                    "b1243" => "4"
                ),
                "b125" => "3",
                "b126" => "3",
            ),
            "b13" => "2",
            "b14" => "2"
        ),
        "c" => "1",
        "d" => "1",
        "e" => "1",
        "f" => "1"
    );
    
    echo (arrayPrinter($array));

?>
